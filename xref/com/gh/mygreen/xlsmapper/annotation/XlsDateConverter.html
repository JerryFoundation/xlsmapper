3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   import java.text.SimpleDateFormat;
9   
10  
11  /**
12   * 日時に対する変換規則を指定するアノテーション。
13   * <p>対応するJavaの型は次の通り。</p>
14   * <ul>
15   *   <li>{@link java.util.Date}</li>
16   *   <li>{@link java.sql.Date}/{@link java.sql.Time}/{@link java.sql.Timestamp}</li>
17   *   <li>{@link java.util.Calendar}</li>
18   * </ul>
19   * 
20   * <h3 class="description">読み込み時の書式の指定</h3>
21   * <p>読み込み時にセルの種類が日付、時刻ではない場合、
22   *    文字列として値を取得し、その値を属性{@link #javaPattern()}で指定した書式に従いパースし、Javaの日時型に変換します。
23   * </p>
24   * 
25   * <ul>
26   *   <li>属性{@link #javaPattern()}で書式を指定します。
27   *       <br>Javaのクラス{@link SimpleDateFormat}で解釈可能な書式を指定します。
28   *   </li>
29   *   <li>属性{@link #locale()}で、ロケールを指定します。
30   *       <br>言語コードのみを指定する場合、'ja'の2桁で指定します。
31   *       <br>言語コードと国コードを指定する場合、'ja _JP'のようにアンダーバーで区切り指定します。
32   *   </li>
33   *   <li>属性{@link #lenient()}で、日付/時刻の解析を厳密に行わないか指定します。
34   *       <br>デフォルト値はtrueで、厳密に解析を行いません。falseの場合厳密に解析を行います。
35   *   </li>
36   *   <li>書式に合わない値をパースした場合、例外{@link com.gh.mygreen.xlsmapper.cellconvert.TypeBindException}が発生します。</li>
37   * </ul>
38   * 
39   * <pre class="highlight"><code class="java">
40   * public class SampleRecord {
41   * 
42   *     {@literal @XlsColumn(columnName="有効期限")}
43   *     {@literal @XlsDateConverter(javaPattern="yyyy年MM月dd日 HH時mm分ss秒", locale="ja_JP", lenient=true)}
44   *     private Date expired;
45   * 
46   * }
47   * </code></pre>
48   * 
49   * <p>アノテーション{@link XlsDateConverter}を付与しない場合、Javaの型ごとに次の書式が標準で適用されます。</p>
50   * <ul>
51   *   <li>{@link java.util.Date}の場合、{@literal "yyyy-MM-dd HH:mm:ss"}</li>
52   *   <li>{@link java.sql.Date}の場合、{@literal "yyyy-MM-dd HH:mm:ss"}</li>
53   *   <li>{@link java.sql.Time}の場合、{@literal "HH:mm:ss"}</li>
54   *   <li>{@link java.sql.Timestamp}の場合、{@literal "yyyy-MM-dd HH:mm:ss.SSS"}</li>
55   *   <li>{@link java.util.Calendar}の場合、{@literal "yyyy-MM-dd HH:mm:ss"}</li>
56   * </ul>
57   * 
58   * <h3 class="description">書き込み時の書式の指定</h3>
59   * <p>書き込み時の書式は、基本的にテンプレートファイルに設定してある値を使用します。
60   *   <br>また、アノテーションでも直接指定することができます。
61   * </p>
62   * 
63   * <ul>
64   *   <li>属性{@link #excelPattern()}で書式を指定します。
65   *       <br>Excelの書式を指定する場合は、
66   *           <a href="http://mygreen.github.io/excel-cellformatter/sphinx/format_basic.html" target="_blank">ユーザ定義</a>
67   *           の形式で指定します。
68   *   </li>
69   * </ul>
70   * 
71   * <pre class="highlight"><code class="java">
72   * public class SampleRecord {
73   * 
74   *     {@literal @XlsColumn(columnName="有効期限")}
75   *     {@literal @XlsDateConverter(excelPattern="[$-411]yyyy\"年\"mm\"月\"dd\"日\" hh\"時\"mm\"分\"ss\"秒\"")}
76   *     private Date expired;
77   * 
78   * }
79   * </code></pre>
80   * 
81   * 
82   * <h3 class="description">書き込み時の注意事項</h3>
83   * 
84   * <p>テンプレートファイルのセルの書式を「標準」に設定している場合に書き込むと、
85   *     書式が「標準」設定の全てのセルの書式が書き換わってしまいます。
86   *   <br>そのため、日付や数値などの書式が必要な場合は、テンプレートファイルで予め書式を設定しておくか、
87   *     アノテーションの属性{@link #excelPattern()}で書式を指定しておいてください。
88   * </p>
89   * 
90   * @version 1.1
91   * @author T.TSUCHIE
92   *
93   */
94  @Target({ElementType.METHOD, ElementType.FIELD})
95  @Retention(RetentionPolicy.RUNTIME)
96  @Documented
97  public @interface XlsDateConverter {
98      
99      /**
100      * 日時の書式パターン。{@link java.text.SimpleDateFormat}の書式を指定します。
101      * <p>読み込み時に、ExcelとJavaの型が一致しない場合に文字列としてパースする際に利用します。</p>
102      * @since 1.1
103      * @return
104      */
105     String javaPattern() default "";
106     
107     /**
108      * 日付／時刻の解析を厳密に行うか指定します。
109      * <p>属性{@link #javaPattern()}を設定した場合に有効になります。</p>
110      * @return
111      */
112     boolean lenient() default false;
113     
114     /**
115      * ロケールの指定を行います。指定しない場合、デフォルトのロケールで処理されます。
116      * <p>例. 'ja_JP', 'ja'</p>
117      * <p>属性{@link #javaPattern()}を設定した場合に有効になります。</p>
118      * @return
119      */
120     String locale() default "";
121     
122     /**
123      * Excelの書式のパターン。
124      * <p>書き込み時に、セルの書式を直接設定したい場合に指定します。</p>
125      * <p>値を指定しない場合、テンプレートファイルに設定してある書式を利用します。</p>
126      * @since 1.1
127      */
128     String excelPattern() default "";
129     
130 }
