3   import java.util.HashMap;
4   import java.util.Map;
5   
6   
7   /**
8    * 式言語の共通インタフェース。
9    * 
10   * @version 0.5
11   * @author T.TSUCHIE
12   *
13   */
14  public abstract class ExpressionLanguage {
15      
16      /**
17       * 引数で与えた式を評価する。
18       * @param expression 評価対象の式。
19       * @param values 式中で利用可のな変数。
20       * @return 評価した式。
21       * @throws ExpressionEvaluationException 式のパースや評価に失敗した場合にスローされます。
22       */
23      public abstract Object evaluate(String expression, Map<String, ?> values);
24      
25      /**
26       * プロパティの値を取得する。
27       * @since 0.5
28       * @param property プロパティの名前
29       * @param target プロパティの取得対象のオブジェクト。
30       * @return プロパティの値。
31       * @throws ExpressionEvaluationException 式のパースや評価に失敗した場合にスローされます。
32       */
33      public Object getProperty(String property, Object target) {
34          
35          final Map<String, Object> vars = new HashMap<>();
36          vars.put("targetObj", target);
37          return evaluate("targetObj." + property, vars);
38          
39      }
40      
41  }
