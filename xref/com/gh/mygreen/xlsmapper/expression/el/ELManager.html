1   package com.gh.mygreen.xlsmapper.expression.el;
2   
3   import java.lang.reflect.Method;
4   
5   import javax.el.ELContext;
6   import javax.el.ExpressionFactory;
7   import javax.el.ValueExpression;
8   
9   
10  /**
11   * EL式を評価するための{@link ExpressionFactory}と{@link ELContext}のインスタンスを管理するクラス。
12   *
13   * @author T.TSUCHIE
14   *
15   */
16  public class ELManager {
17      
18      private LocalELContext elContext;
19      
20      /**
21       * {@link ExpressionFactory}のインスタンスを取得する。
22       * @return
23       */
24      public static ExpressionFactory getExpressionFactory() {
25          return ExpressionFactory.newInstance();
26      }
27      
28      /**
29       * {@link ELContext}のインスタンスを取得する。{@link LocalELContext}を返す。
30       * @return
31       */
32      public LocalELContext getELContext() {
33          if(elContext == null) {
34              this.elContext = new LocalELContext();
35          }
36          
37          return elContext;
38      }
39      
40      /**
41       * EL関数を登録する。
42       * @param prefix
43       * @param function
44       * @param method
45       */
46      public void mapFunction(String prefix, String function, Method method) {
47          getELContext().mapFunction(prefix, function, method);
48      }
49      
50      /**
51       * EL式中の変数を登録する。
52       * @param variable
53       * @param expression
54       */
55      public void setVariable(final String variable, final ValueExpression expression) {
56          getELContext().setVariable(variable, expression);
57      }
58  }
