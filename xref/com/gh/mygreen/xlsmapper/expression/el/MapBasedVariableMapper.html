3   import java.util.Collections;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   import javax.el.ValueExpression;
8   import javax.el.VariableMapper;
9   
10  
11  /**
12   * 変数のマッパーを作成する
13   *
14   * @author T.TSUCHIE
15   *
16   */
17  public class MapBasedVariableMapper extends VariableMapper {
18      
19      private Map<String, ValueExpression> varMap = Collections.emptyMap();
20      
21      @Override
22      public ValueExpression resolveVariable(final String variable) {
23          return varMap.get(variable);
24      }
25      
26      @Override
27      public ValueExpression setVariable(final String variable, final ValueExpression expression) {
28          if(varMap.isEmpty()) {
29              varMap = new HashMap<String, ValueExpression>();
30          }
31          
32          return varMap.put(variable, expression);
33      }
34  }
