1   package com.gh.mygreen.xlsmapper.expression.el;
2   
3   import java.util.Formatter;
4   import java.util.Locale;
5   
6   /**
7    * EL式中で{@code java.util.Formatter#format}を使用すると、オーバーロードしているメソッドと区別がつかなくなるため、
8    * ラップしオーバーロードのメソッドをなくすため利用します。
9    *
10   */
11  public class FormatterWrapper {
12      
13      private final Formatter formatter;
14      
15      public FormatterWrapper(final Formatter formatter) {
16          this.formatter = formatter;
17      }
18      
19      public FormatterWrapper(final Locale locale) {
20          this.formatter = new Formatter(locale);
21      }
22      
23      public String format(final String format, final Object... args) {
24          return formatter.format(format, args).toString();
25      }
26      
27      @Override
28      public String toString() {
29          final StringBuilder sb = new StringBuilder();
30          sb.append("FormatterWrapper");
31          sb.append("{}");
32          return sb.toString();
33      }
34  }
