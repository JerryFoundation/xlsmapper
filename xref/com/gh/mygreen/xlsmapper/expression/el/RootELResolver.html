1   package com.gh.mygreen.xlsmapper.expression.el;
2   
3   import java.beans.FeatureDescriptor;
4   import java.util.Collections;
5   import java.util.IllegalFormatException;
6   import java.util.Iterator;
7   import java.util.Map;
8   import java.util.concurrent.ConcurrentHashMap;
9   
10  import javax.el.ELContext;
11  import javax.el.ELException;
12  import javax.el.ELResolver;
13  import javax.el.PropertyNotFoundException;
14  
15  /**
16   * ルートノードを処理するためのELResolver.
17   * <p>ELはルートノードのアクセス時のみgetValueのbaseがnullとなりますので、
18   *    その場合のみ指定のオブジェクトを探索するようにし、それ以外はbaseを利用するようにするという程度のものです
19   *
20   * @author T.TSUCHIE
21   *
22   */
23  public class RootELResolver extends ELResolver {
24      
25      /**
26       * EL式中で利用可能な標準のフォーマッターの変数の名称。
27       * Name under which to bind a formatter to the EL context.
28       */
29      public static final String FORMATTER = "formatter";
30      private static final String FORMAT = "format";
31      
32      /**
33       * プロパティ（変数）の値のキャッシュ。
34       */
35      private final Map<String, Object> map = new ConcurrentHashMap<String, Object>();
36      
37      /**
38       * 式中の変数が解決できない場合無視するかどうか。
39       */
40      private final boolean ignoreNotFoundProperty;
41      
42      public RootELResolver() {
43          this(false);
44      }
45      
46      /**
47       * 式中のプロパティが見つからない場合無視するかどうかを指定するコンストラクタ。
48       * @param ignoreNotFoundProperty 式中のプロパティが見つからない場合無視するかどうか
49       */
50      public RootELResolver(final boolean ignoreNotFoundProperty) {
51          this.ignoreNotFoundProperty = ignoreNotFoundProperty;
52      }
53      
54      
55      /**
56       * {@inheritDoc}
57       * <p>baseがnull、つまり<code>${first.xxxx}</code>のfirstの場合は、文字列として変数名を受け取ることを示す。
58       */
59      @Override
60      public Class<?> getCommonPropertyType(final ELContext context, final Object base) {
61          return base == null ? String.class : null;
62      }
63      
64      @Override
65      public Iterator<FeatureDescriptor> getFeatureDescriptors(final ELContext context, final Object base) {
66          
67          return base == null ? null : Collections.<FeatureDescriptor>emptyListIterator();
68      }
69      
70      @Override
71      public Class<?> getType(final ELContext context, final Object base, final Object property) {
72          // nulの場合、次のELResolverへ
73          return resolve(context, base, property) ? Object.class : null;
74      }
75      
76      @Override
77      public Object getValue(final ELContext context, final Object base, final Object property) {
78          if(resolve(context, base, property)) {
79              if(!isProperty((String) property)) {
80                  if(isIgnoreNotFoundProperty()) {
81                      return null;
82                  }
83                  throw new PropertyNotFoundException("Cannot find property " + property);
84              }
85              return getProperty((String) property);
86          }
87          
88          // 不明な場合は次のELResolverへ
89          return null;
90      }
91      
92      @Override
93      public void setValue(final ELContext context, final Object base, final Object property, final Object value) {
94          if(resolve(context, base, property)) {
95              setProperty((String) property, value);
96          }
97      }
98      
99      @Override
100     public boolean isReadOnly(final ELContext context, final Object base, final Object property) {
101         return true;
102     }
103     
104     @Override
105     public Object invoke(final ELContext context, final Object base, final Object method,
106             final Class<?>[] paramTypes, final Object[] params) {
107         
108         if(resolve(context, base, method)) {
109             throw new ELException("Invalid property");
110         }
111         Object returnValue = null;
112         // due to bugs in most EL implementations when it comes to evaluating varargs we take care
113         // of the formatter call
114         // ourselves.
115         if(base instanceof FormatterWrapper) {
116             returnValue = evaluateFormatExpression(context, method, params);
117         }
118         return returnValue;
119     }
120     
121     private Object evaluateFormatExpression(final ELContext context, final Object method, final Object[] params) {
122         if(!FORMAT.equals(method)) {
123             throw new ELException(String.format("Wrong method name 'formatter#%s' does not exist. Only formatter#format is supported.", method));
124         }
125         
126         if(params.length == 0) {
127             throw new ELException("Invalid number of arguments to Formatter#format");
128         }
129         if(!(params[0] instanceof String)) {
130             throw new ELException("The first argument to Formatter#format must be String");
131         }
132         
133         final FormatterWrapper formatterWrapper = (FormatterWrapper)context.getVariableMapper()
134                 .resolveVariable(FORMATTER)
135                 .getValue(context);
136         final Object[] formattingParameters = new Object[params.length - 1];
137         System.arraycopy( params, 1, formattingParameters, 0, params.length - 1 );
138         
139         Object returnValue;
140         try {
141             returnValue = formatterWrapper.format( (String) params[0], formattingParameters );
142             context.setPropertyResolved(true);
143         } catch (IllegalFormatException e) {
144             throw new ELException("Error in Formatter#format call", e);
145         }
146         
147         return returnValue;
148     }
149     
150     private boolean resolve(final ELContext context, final Object base, final Object property) {
151         context.setPropertyResolved( base == null && property instanceof String );
152         // 解釈済みであることを示す
153         return context.isPropertyResolved();
154     }
155     
156     private Object getProperty(String property) {
157         return map.get(property);
158     }
159     
160     private void setProperty(String property, Object value) {
161         map.put(property, value);
162     }
163     
164     private boolean isProperty(String property) {
165         return map.containsKey(property);
166     }
167     
168     /**
169      * 式中のプロパティ（変数）が見つからない場合、無視するかどうか。
170      * <p>無視しない場合、例外{@link PropertyNotFoundException}が発生します。
171      * @return
172      */
173     public boolean isIgnoreNotFoundProperty() {
174         return ignoreNotFoundProperty;
175     }
176     
177 }
