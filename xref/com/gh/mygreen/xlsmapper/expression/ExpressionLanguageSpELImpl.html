18  /**
19   * SpELを使用するための実装。
20   * <p>利用するには、Spring3で追加されたライブラリ「spring-expression」が必要。
21   * 
22   */
23  public class ExpressionLanguageSpELImpl implements ExpressionLanguage {
24      
25      private static final Logger logger = LoggerFactory.getLogger(ExpressionLanguageSpELImpl.class);
26      
27      private final ObjectCache<String, Object> expressionCache = new ObjectCache<String, Object>();
28      
29      /**
30       * {@inheritDoc}
31       */
32      @Override
33      public Object evaluate(final String expression, final Map<String, ?> values) {
34          
35          ArgUtils.notEmpty(expression, "expression");
36          ArgUtils.notEmpty(values, "values");
37          
38          try {
39              final EvaluationContext ctx = new StandardEvaluationContext();
40              for (final Entry<String, ? > entry : values.entrySet()) {
41                  ctx.setVariable(entry.getKey(), entry.getValue());
42              }
43              
44              if(logger.isDebugEnabled()) {
45                  logger.debug("Evaluating SpEL expression: {}", expression);
46              }
47              
48              Expression expr = (Expression) expressionCache.get(expression);
49              if (expr == null) {
50                  ExpressionParser parser = new SpelExpressionParser();
51                  expr = parser.parseExpression(expression);
52                  expressionCache.put(expression, expr);
53              }
54              
55              return expr.getValue(ctx);
56          
57          } catch (final ExpressionException ex){
58              throw new ExpressionEvaluationException(String.format("Evaluating [%s] script with SpEL failed.", expression), ex);
59          }
60      }
61      
62  }
