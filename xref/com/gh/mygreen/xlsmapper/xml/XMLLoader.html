18  /**
19   * Loads annotation informations as {@link com.gh.mygreen.xlsmapper.xml.XMLInfo} 
20   * from the XML file.
21   * 
22   * @author Naoki Takezoe
23   */
24  public class XMLLoader {
25  
26  	public static XMLInfo load(InputStream in) throws XmlLoadException {
27  		try {
28  			XMLInfo info = new XMLInfo();
29  			
30  			DocumentBuilderFactory dbfactory = DocumentBuilderFactory.newInstance();
31  			DocumentBuilder builder = dbfactory.newDocumentBuilder();
32  			Document doc = builder.parse(in);
33  			
34  			Element root = doc.getDocumentElement();
35  			NodeList classes = root.getElementsByTagName("class");
36  			for(int i=0;i<classes.getLength();i++){
37  				Element clazz = (Element)classes.item(i);
38  				String className = clazz.getAttribute("name");
39  				if(className==null || className.length()==0){
40  					throw new XmlLoadException("'name' attribute is required.");
41  				}
42  				AnnotationInfo[] classAnns =
43  					getAnnotationInfo(clazz.getElementsByTagName("annotation"));
44  				
45  				ClassInfo classInfo = new ClassInfo();
46  				classInfo.setClassName(className);
47  				for(AnnotationInfo annInfo: classAnns){
48  					classInfo.addAnnotationInfo(annInfo);
49  				}
50  				
51  				NodeList methods = clazz.getElementsByTagName("method");
52  				for(int j=0;j<methods.getLength();j++){
53  					Element method = (Element)methods.item(j);
54  					String methodName = method.getAttribute("name");
55  					if(methodName==null || methodName.length()==0){
56  						throw new XmlLoadException("'name' attribute is required.");
57  					}
58  					AnnotationInfo[] methodAnns =
59  						getAnnotationInfo(method.getElementsByTagName("annotation"));
60  					
61  					MethodInfo methodInfo = new MethodInfo();
62  					methodInfo.setMethodName(methodName);
63  					for(AnnotationInfo annInfo: methodAnns){
64  						methodInfo.addAnnotationInfo(annInfo);
65  					}
66  					
67  					classInfo.addMethodInfo(methodInfo);
68  				}
69  				
70  				NodeList fields = clazz.getElementsByTagName("field");
71  				for(int j=0;j<fields.getLength();j++){
72  					Element field = (Element)fields.item(j);
73  					String fieldName = field.getAttribute("name");
74  					if(fieldName==null || fieldName.length()==0){
75  						throw new XmlLoadException("'name' attribute is required.");
76  					}
77  					AnnotationInfo[] fieldAnns =
78  						getAnnotationInfo(field.getElementsByTagName("annotation"));
79  					
80  					FieldInfo fieldInfo = new FieldInfo();
81  					fieldInfo.setFieldName(fieldName);
82  					for(AnnotationInfo annInfo: fieldAnns){
83  						fieldInfo.addAnnotationInfo(annInfo);
84  					}
85  					
86  					classInfo.addFieldInfo(fieldInfo);
87  				}
88  				
89  				info.addClassInfo(classInfo);
90  			}
91  			
92  			return info;
93  			
94  		} catch(XmlLoadException ex){
95  			throw ex;
96  		} catch(Exception ex){
97  			throw new XmlLoadException(ex);
98  		}
99  	}

101 	private static AnnotationInfo[] getAnnotationInfo(NodeList annotations) throws XmlLoadException {
102 		List<AnnotationInfo> list = new ArrayList<AnnotationInfo>();
103 		for(int i=0;i<annotations.getLength();i++){
104 			Element annotation = (Element)annotations.item(i);
105 			String className = annotation.getAttribute("name");
106 			if(className==null || className.length()==0){
107 				throw new XmlLoadException("'name' attribute is required.");
108 			}
109 			
110 			AnnotationInfo info = new AnnotationInfo();
111 			info.setAnnotationClass(className);
112 			
113 			Map<String, String> attrs = getAttributes(annotation);
114 			for(Map.Entry<String, String> entry: attrs.entrySet()){
115 				info.addAnnotationAttribute(entry.getKey(), entry.getValue());
116 			}
117 			
118 			list.add(info);
119 		}
120 		return list.toArray(new AnnotationInfo[list.size()]);
121 	}
122 	
123 	private static Map<String, String> getAttributes(Element annotations) throws XmlLoadException {
124 		Map<String, String> map = new HashMap<String, String>();
125 		NodeList attrs = annotations.getElementsByTagName("attribute");
126 		for(int i=0;i<attrs.getLength();i++){
127 			Element attr = (Element)attrs.item(i);
128 			String name = attr.getAttribute("name");
129 			String value = getChildText(attr);
130 			
131 			if(name==null || name.length()==0){
132 				throw new XmlLoadException("'name' attribute is required.");
133 			}
134 			
135 			map.put(name, value);
136 		}
137 		return map;
138 	}
139 	
140 	private static String getChildText(Element element){
141 		StringBuffer sb = new StringBuffer();
142 		NodeList children = element.getChildNodes();
143 		for(int i=0;i<children.getLength();i++){
144 			Node node = children.item(i);
145 			if(node instanceof Text){
146 				sb.append(node.getNodeValue());
147 			}
148 		}
149 		return sb.toString().trim();
150 	}
