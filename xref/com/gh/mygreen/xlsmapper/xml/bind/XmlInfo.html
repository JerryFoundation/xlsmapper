3   import java.io.Serializable;
4   import java.util.LinkedHashMap;
5   import java.util.Map;
6   
7   import javax.xml.bind.annotation.XmlElement;
8   import javax.xml.bind.annotation.XmlRootElement;
9   
10  import com.gh.mygreen.xlsmapper.ArgUtils;
11  
12  
13  /**
14   * XMLで定義したアノテーションの設定情報。
15   * 
16   * <pre>
17   *   XMLの使用：
18   *   <annotations> <- ルート要素
19   *     
20   *   </annotations>
21   * 
22   * </pre>
23   * 
24   * @since 0.5
25   * @author T.TSUCHIE
26   * 
27   */
28  @XmlRootElement(name="annotations")
29  public class XmlInfo implements Serializable {
30      
31      /** serialVersionUID */
32      private static final long serialVersionUID = 1L;
33      
34      /**
35       * クラス名をキーとしたクラス情報のマップ
36       */
37      private Map<String, ClassInfo> classInfos = new LinkedHashMap<>();
38      
39      /**
40       * クラス情報を追加する。
41       * @param classInfo
42       */
43      @XmlElement(name="class")
44      public void setClassInfo(final ClassInfo classInfo) {
45          ArgUtils.notNull(classInfo, "classInfo");
46          
47          this.classInfos.put(classInfo.getClassName(), classInfo);
48      }
49      
50      /**
51       * クラス名を指定してクラス情報を取得する。
52       * @param clazzName クラス名
53       * @return
54       */
55      public ClassInfo getClassInfo(final String clazzName) {
56          return classInfos.get(clazzName);
57      }
58      
59      @Override
60      public String toString() {
61          StringBuilder sb = new StringBuilder();
62          sb.append("XmlInfo:");
63          
64          for(ClassInfo clazz : classInfos.values()) {
65              sb.append("\n  ").append(clazz.toString());
66          }
67          
68          return sb.toString();
69      }
70      
71  }
