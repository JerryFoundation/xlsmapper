3   import java.lang.annotation.Annotation;
4   
5   import com.gh.mygreen.xlsmapper.XlsMapperException;
6   import com.gh.mygreen.xlsmapper.annotation.converter.XlsConverter;

8   import com.gh.mygreen.xlsmapper.cellconvert.CellConverterRegistry;
9   import com.gh.mygreen.xlsmapper.cellconvert.ConversionException;
10  import com.gh.mygreen.xlsmapper.cellconvert.DefaultCellConverter;
11  
12  
13  /**
14   * 各種アノテーションを処理するためのクラスの抽象クラス。
15   * <p>通常はこのクラスを継承して作成する。
16   * 
17   * @author T.TSUCHIE
18   *
19   */
20  public abstract class AbstractFieldProcessor<A extends Annotation> implements LoadingFieldProcessor<A>, SavingFieldProcessor<A> {
21      
22      /**
23       * 該当するタイプのConverterが見つからないときの例外のインスタンスを作成する。
24       * @param targetType クラスタイプ
25       * @return
26       */
27      protected ConversionException newNotFoundConverterExpcetion(final Class<?> targetType) {
28          return new ConversionException(
29                  String.format("not found CellConverter for type '%s'.", targetType.getName()),
30                  targetType);
31      }
32      
33      /**
34       * 読み込み時用のConveterを取得する。
35       * <p>アノテーション「{@link XlsConverter#converterClass()}」が設定されていた場合を考慮した、個別のConverterを考慮する。
36       * 
37       * @param adaptor
38       * @param converterResolver
39       * @return
40       * @throws XlsMapperException Converterが見つからない場合。
41       */
42      protected CellConverter<?> getLoadingCellConverter(final FieldAdaptor adaptor, final CellConverterRegistry converterResolver) throws XlsMapperException {
43          
44          final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
45          final CellConverter<?> converter;
46          
47          if(converterAnno != null && !converterAnno.converterClass().equals(DefaultCellConverter.class)) {
48              converter = converterResolver.createCellConverter((Class<CellConverter>) converterAnno.converterClass());
49          } else {
50              converter = converterResolver.getConverter(adaptor.getTargetClass());
51              if(converter == null) {
52                  throw newNotFoundConverterExpcetion(adaptor.getTargetClass());
53              }
54          }
55          
56          return converter;
57      }
58      
59      /**
60       * 書き込み時用のConveterを取得する。
61       * <p>アノテーション「{@link XlsConverter#converterClass()}」が設定されていた場合を考慮した、個別のConverterを考慮する。
62       * 
63       * @param adaptor
64       * @param converterResolver
65       * @return
66       * @throws XlsMapperException Converterが見つからない場合。
67       */
68      protected CellConverter<?> getSavingCellConverter(final FieldAdaptor adaptor, final CellConverterRegistry converterResolver) throws XlsMapperException {
69          
70          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
71          final CellConverter<?> converter;
72          
73          if(converterAnno != null && !converterAnno.converterClass().equals(DefaultCellConverter.class)) {
74              try {
75                  converter = converterAnno.converterClass().newInstance();
76              } catch (InstantiationException | IllegalAccessException e) {
77                  throw new XlsMapperException(String.format("fail create '%s' instance.", converterAnno.converterClass().getName()),
78                          e);
79              }
80          } else {
81              converter = converterResolver.getConverter(adaptor.getTargetClass());
82              if(converter == null) {
83                  throw newNotFoundConverterExpcetion(adaptor.getTargetClass());
84              }
85          }
86          
87          return converter;
88      }
89  }
