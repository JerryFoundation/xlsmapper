12  public class CellNotFoundException extends XlsMapperException {
13      
14      /** serialVersionUID */
15      private static final long serialVersionUID = -3913407241079675756L;
16      
17      /** 検索元のシート */
18      private final String sheetName;
19      
20      /** 検索対象のセルのラベル */
21      private final String label;
22      
23      public CellNotFoundException(final String sheetName, final String label) {
24          super(String.format("Cell '%s' not found in sheet '%s'", label, sheetName));
25          this.sheetName = sheetName;
26          this.label = label;
27          
28      }
29      
30      public String getSheetName() {

32      }
33
