3   import java.lang.annotation.Annotation;
4   import java.util.Map;
5   import java.util.concurrent.ConcurrentHashMap;
6   
7   import com.gh.mygreen.xlsmapper.ArgUtils;
8   import com.gh.mygreen.xlsmapper.annotation.XlsCell;
9   import com.gh.mygreen.xlsmapper.annotation.XlsHorizontalRecords;
10  import com.gh.mygreen.xlsmapper.annotation.XlsIterateTables;
11  import com.gh.mygreen.xlsmapper.annotation.XlsLabelledCell;
12  import com.gh.mygreen.xlsmapper.annotation.XlsSheetName;
13  import com.gh.mygreen.xlsmapper.annotation.XlsVerticalRecords;
14  import com.gh.mygreen.xlsmapper.fieldprocessor.processor.CellProcessor;
15  import com.gh.mygreen.xlsmapper.fieldprocessor.processor.HorizontalRecordsProcessor;
16  import com.gh.mygreen.xlsmapper.fieldprocessor.processor.IterateTablesProcessor;
17  import com.gh.mygreen.xlsmapper.fieldprocessor.processor.LabelledCellProcessor;
18  import com.gh.mygreen.xlsmapper.fieldprocessor.processor.SheetNameProcessor;
19  import com.gh.mygreen.xlsmapper.fieldprocessor.processor.VerticalRecordsProcessor;

22  /**
23   * {@link FieldProcessor}を管理するクラス。
24   * 
25   * @version 1.0
26   * @author Naoki Takezoe
27   * @author T.TSUCHIE
28   *
29   */
30  public class FieldProcessorRegstry {
31      
32      private Map<Class<? extends Annotation>, LoadingFieldProcessor<?>> loadingPocessorMap;
33      
34      private Map<Class<? extends Annotation>, SavingFieldProcessor<?>> savingPocessorMap;
35      
36      public FieldProcessorRegstry() {
37          this.loadingPocessorMap = new ConcurrentHashMap<>();
38          
39          init();
40      }
41      
42      /**
43       * 初期化する。
44       * <p>アノテーションとプロセッサの初期化を行う。
45       */
46      protected void init() {
47          
48          if(loadingPocessorMap == null) {
49              this.loadingPocessorMap = new ConcurrentHashMap<Class<? extends Annotation>, LoadingFieldProcessor<?>>();
50          } else {
51              loadingPocessorMap.clear();
52          }
53          
54          if(savingPocessorMap == null) {
55              this.savingPocessorMap = new ConcurrentHashMap<Class<? extends Annotation>, SavingFieldProcessor<?>>();
56          } else {
57              savingPocessorMap.clear();
58          }
59          
60          //標準のフィールドプロセッサを登録する。
61          registerProcessor(XlsSheetName.class, new SheetNameProcessor());
62          registerProcessor(XlsCell.class, new CellProcessor());
63          registerProcessor(XlsLabelledCell.class, new LabelledCellProcessor());
64          registerProcessor(XlsHorizontalRecords.class, new HorizontalRecordsProcessor());
65          registerProcessor(XlsVerticalRecords.class, new VerticalRecordsProcessor());
66          registerProcessor(XlsIterateTables.class, new IterateTablesProcessor());
67          
68      }
69      
70      /**
71       * アノテーションに対する{@link LoadingFieldProcessor}を取得する。
72       * @param anno 取得対象のアノテーションのインスタンス。
73       * @return 見つからない場合はnullを返す。
74       */
75      @SuppressWarnings("unchecked")
76      public <A extends Annotation> LoadingFieldProcessor<A> getLoadingProcessor(final Annotation anno) {
77          ArgUtils.notNull(anno, "anno");
78          
79          return (LoadingFieldProcessor<A>) loadingPocessorMap.get(anno.annotationType());
80      }
81      
82      /**
83       * アノテーションに対する{@link LoadingFieldProcessor}を取得する。
84       * @param annoClass  取得対象のアノテーションのクラスタイプ。
85       * @return 見つからない場合はnullを返す。
86       */
87      @SuppressWarnings("unchecked")
88      public <A extends Annotation> LoadingFieldProcessor<A> getLoadingProcessor(final Class<A> annoClass) {
89          ArgUtils.notNull(annoClass, "annoClass");
90          
91          return (LoadingFieldProcessor<A>) loadingPocessorMap.get(annoClass);
92      }
93      
94      /**
95       * アノテーションに対する{@link SavingFieldProcessor}を取得する。
96       * @param anno 取得対象のアノテーションのインスタンス。
97       * @return 見つからない場合はnullを返す。
98       */
99      @SuppressWarnings("unchecked")
100     public <A extends Annotation> SavingFieldProcessor<A> getSavingProcessor(final Annotation anno) {
101         ArgUtils.notNull(anno, "anno");
102         
103         return (SavingFieldProcessor<A>) getSavingProcessor(anno.annotationType());
104     }
105     
106     /**
107      * アノテーションに対する{@link SavingFieldProcessor}を取得する。
108      * @param annoClass 取得対象のアノテーションのクラスタイプ。
109      * @return 見つからない場合はnullを返す。
110      */
111     @SuppressWarnings("unchecked")
112     public <A extends Annotation> SavingFieldProcessor<A> getSavingProcessor(final Class<A> annoClass) {
113         ArgUtils.notNull(annoClass, "annoClass");
114         
115         return (SavingFieldProcessor<A>) savingPocessorMap.get(annoClass);
116     }
117     
118     /**
119      * アノテーションに対する{@link FieldProcessor}を登録する。
120      * @param annoClass 登録対象のアノテーションのクラスタイプ。
121      * @param processor フィールドプロセッサーのインスタンス。
122      *                  {@link LoadingFieldProcessor}または{@link SavingFieldProcessor} を実装している必要がある。
123      */
124     public <A extends Annotation> void registerProcessor(final Class<A> annoClass, final FieldProcessor<A> processor) {
125         ArgUtils.notNull(annoClass, "annoClass");
126         ArgUtils.notNull(processor, "processor");
127         
128         if(processor instanceof LoadingFieldProcessor) {
129             loadingPocessorMap.put(annoClass, (LoadingFieldProcessor) processor);
130         }
131         
132         if(processor instanceof SavingFieldProcessor) {
133             savingPocessorMap.put(annoClass, (SavingFieldProcessor) processor);
134         }
135     }
136     
137 }
