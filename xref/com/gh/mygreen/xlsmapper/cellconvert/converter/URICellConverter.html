28  public class URICellConverter extends AbstractCellConverter<URI> {
29      
30      @Override
31      public URI toObject(final Cell cell, final FieldAdaptor adaptor, final XlsMapperConfig config)
32              throws XlsMapperException {
33          
34          final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
35          
36          if(POIUtils.isEmptyCellContents(cell, config.getCellFormatter())) {
37              
38              if(Utils.hasNotDefaultValue(converterAnno)) {
39                  return null;
40              } else {
41                  final String defaultValue = converterAnno.defaultValue();
42                  try {
43                      return new URI(defaultValue);
44                  } catch (URISyntaxException e) {
45                      throw newTypeBindException(e, cell, adaptor, defaultValue);
46                  }
47              }
48              
49          } else if(cell.getHyperlink() != null) {
50              // リンクが設定されているセルは、リンクの内容を値とする
51              final String address = Utils.trim(cell.getHyperlink().getAddress(), converterAnno);
52              try {
53                  return new URI(address);
54              } catch (URISyntaxException e) {
55                  throw newTypeBindException(e, cell, adaptor, address);
56              }
57              
58          } else {
59              // リンクがないセルは、セルの文字列を値とする
60              final String str = Utils.trim(POIUtils.getCellContents(cell, config.getCellFormatter()), converterAnno);
61              if(Utils.isEmpty(str)) {
62                  return null;
63              }
64              
65              try {
66                  return new URI(str);
67              } catch (URISyntaxException e) {
68                  throw newTypeBindException(cell, adaptor, str);
69              }
70              
71          }
72      }
73      
74      @Override
75      public Cell toCell(final FieldAdaptor adaptor, final URI targetValue, final Object targetBean,
76              final Sheet sheet, final int column, final int row,
77              final XlsMapperConfig config) throws XlsMapperException {
78          
79          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
80          final XlsFormula formulaAnno = adaptor.getSavingAnnotation(XlsFormula.class);
81          final boolean primaryFormula = formulaAnno == null ? false : formulaAnno.primary();
82          
83          final Cell cell = POIUtils.getCell(sheet, column, row);
84          
85          // セルの書式設定
86          if(converterAnno != null) {
87              POIUtils.wrapCellText(cell, converterAnno.wrapText());
88              POIUtils.shrinkToFit(cell, converterAnno.shrinkToFit());
89          }
90          
91          URI value = targetValue;
92          
93          // 既存のハイパーリンクを削除
94          // 削除しないと、Excelの見た目上はリンクは変わっているが、データ上は2重にリンクが設定されている。
95          POIUtils.removeHyperlink(cell);
96          
97          if(value != null && !primaryFormula) {
98              
99              final CreationHelper helper = sheet.getWorkbook().getCreationHelper();
100             final Hyperlink link = helper.createHyperlink(Hyperlink.LINK_URL);
101             link.setAddress(value.toString());
102             cell.setHyperlink(link);
103             
104             cell.setCellValue(value.toString());
105             
106         } else if(formulaAnno != null) {
107             Utils.setupCellFormula(adaptor, formulaAnno, config, cell, targetBean);
108             
109         } else {
110             cell.setCellType(Cell.CELL_TYPE_BLANK);
111         }
112         
113         return cell;
114     }
