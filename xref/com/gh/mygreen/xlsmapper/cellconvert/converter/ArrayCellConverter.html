19  /**
20   * 配列型を変換するためのConverter。
21   *
22   * @version 1.0
23   * @author T.TSUCHIE
24   *
25   */
26  public class ArrayCellConverter extends AbstractCellConverter<Object[]> {
27      
28      @Override
29      public Object[] toObject(final Cell cell, final FieldAdaptor adaptor, final XlsMapperConfig config)
30              throws XlsMapperException {
31          
32          final ListCellConverter converter = new ListCellConverter();
33          final XlsArrayConverter anno = converter.getLoadingAnnotation(adaptor);
34          
35          Class<?> itemClass = anno.itemClass();
36          if(itemClass == Object.class) {
37              itemClass = adaptor.getLoadingGenericClassType();
38          }
39          
40          final List<?> list = converter.toObject(cell, adaptor, config);
41          return list.toArray(((Object[])Array.newInstance(itemClass, list.size())));
42      }
43      
44      @Override
45      public Cell toCell(final FieldAdaptor adaptor, final Object[] targetValue, final Sheet sheet, final int column, final int row,
46              final XlsMapperConfig config) throws XlsMapperException {
47          
48          final ListCellConverter converter = new ListCellConverter();
49          
50          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
51          final XlsArrayConverter anno = converter.getSavingAnnotation(adaptor);
52          
53          Class<?> itemClass = anno.itemClass();
54          if(itemClass == Object.class) {
55              itemClass = adaptor.getSavingGenericClassType();
56          }
57          
58          final Cell cell = POIUtils.getCell(sheet, column, row);
59          
60          // セルの書式設定
61          if(converterAnno != null) {
62              POIUtils.wrapCellText(cell, converterAnno.forceWrapText());
63              POIUtils.shrinkToFit(cell, converterAnno.forceShrinkToFit());
64          }
65          
66          Object[] value = targetValue;
67          // デフォルト値から値を設定する
68          if(Utils.isEmpty(value) && Utils.hasDefaultValue(converterAnno)) {
69              final List<?> list = converter.convertList(Utils.getDefaultValue(converterAnno), itemClass, converterAnno, anno);
70              value = list.toArray(((Object[])Array.newInstance(itemClass, list.size())));
71          }
72          
73          if(Utils.isNotEmpty(value)) {
74              final boolean trim = (converterAnno == null ? false : converterAnno.trim()); 
75              final String cellValue = Utils.join(value, anno.separator(), anno.ignoreEmptyItem(), trim);
76              cell.setCellValue(cellValue);
77          } else {
78              cell.setCellType(Cell.CELL_TYPE_BLANK);
79          }
80          
81          return cell;
82      }
