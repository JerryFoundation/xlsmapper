16  /**
17   * char/Charcter型を処理するConverter.
18   * 
19   * @since 1.5
20   * @author T.TSUCHIE
21   *
22   */
23  public class CharacterCellConverter extends AbstractCellConverter<Character> {
24  
25      @Override
26      public Character toObject(Cell cell, FieldAdaptor adaptor, XlsMapperConfig config) {
27          
28          final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
29          
30          String cellValue = POIUtils.getCellContents(cell, config.getCellFormatter());
31          cellValue = Utils.trim(cellValue, converterAnno);
32          
33          Character resultValue = null;
34          if(Utils.isEmpty(cellValue)) {
35              if(Utils.hasDefaultValue(converterAnno)) {
36                  resultValue = converterAnno.defaultValue().charAt(0);
37              }
38              
39          } else {
40              resultValue = cellValue.charAt(0);
41          }
42          
43          if(resultValue == null && adaptor.getTargetClass().isPrimitive()) {
44              resultValue = '\u0000';
45          }
46          
47          return resultValue;
48      }
49  
50      @Override
51      public Cell toCell(final FieldAdaptor adaptor, final Character targetValue, final Object targetBean,
52              final Sheet sheet, final int column, final int row, final XlsMapperConfig config) throws XlsMapperException {
53          
54          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
55          final XlsFormula formulaAnno = adaptor.getSavingAnnotation(XlsFormula.class);
56          final boolean primaryFormula = formulaAnno == null ? false : formulaAnno.primary();
57          
58          final Cell cell = POIUtils.getCell(sheet, column, row);
59          
60          // セルの書式設定
61          if(converterAnno != null) {
62              POIUtils.wrapCellText(cell, converterAnno.wrapText());
63              POIUtils.shrinkToFit(cell, converterAnno.shrinkToFit());
64          }
65          
66          Character value = targetValue;
67          
68          String cellValue = Utils.trim(value, converterAnno);
69          cellValue = Utils.getDefaultValueIfEmpty(cellValue, converterAnno);
70          if(cellValue.length() >= 1) {
71              cellValue = cellValue.substring(0, 1);
72          }
73          
74          if(Utils.isNotEmpty(cellValue) && !primaryFormula) {
75              cell.setCellValue(cellValue);
76              
77          } else if(formulaAnno != null) {
78              Utils.setupCellFormula(adaptor, formulaAnno, config, cell, targetBean);
79              
80          } else {
81              cell.setCellType(Cell.CELL_TYPE_BLANK);
82          }
83          
84          return cell;
85      }
86  }
