15  /**
16   * String型を処理するためのConverter.
17   *
18   * @author T.TSUCHIE
19   *
20   */
21  public class StringCellConverter extends AbstractCellConverter<String> {
22  
23      @Override
24      public String toObject(final Cell cell, final FieldAdaptor adaptor, final XlsMapperConfig config) {
25          
26          final XlsConverter converterAnno = adaptor.getLoadingAnnotation(XlsConverter.class);
27          
28          String resultValue = POIUtils.getCellContents(cell, config.getCellFormatter());
29          resultValue = Utils.trim(resultValue, converterAnno);
30          
31          if(resultValue.isEmpty()) {
32              if(Utils.hasDefaultValue(converterAnno)) {
33                  resultValue = converterAnno.defaultValue();
34                  
35              } else if(Utils.getTrimValue(converterAnno)) {
36                  // trimが有効な場合、空文字を設定する。
37                  resultValue = "";
38                  
39              } else {
40                  resultValue = null;
41              }
42          }
43          
44          return resultValue;
45      }
46      
47      @Override
48      public Cell toCell(final FieldAdaptor adaptor, final Object targetObj, final Sheet sheet, final int column, final int row,
49              final XlsMapperConfig config) {
50          
51          return toCell(adaptor, targetObj, sheet, column, row, config, null);
52      }
53      
54      @Override
55      public Cell toCellWithMap(FieldAdaptor adaptor, String key, Object targetObj, Sheet sheet, int column, int row, XlsMapperConfig config)
56              throws XlsMapperException {
57          
58          return toCell(adaptor, targetObj, sheet, column, row, config, key);
59      }
60      
61      private Cell toCell(final FieldAdaptor adaptor, final Object targetObj, final Sheet sheet, final int column, final int row,
62              final XlsMapperConfig config, final String mapKey) {
63          
64          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
65          
66          final Cell cell = POIUtils.getCell(sheet, column, row);
67          
68          // セルの書式設定
69          if(converterAnno != null) {
70              POIUtils.wrapCellText(cell, converterAnno.forceWrapText());
71              POIUtils.shrinkToFit(cell, converterAnno.forceShrinkToFit());
72          }
73          
74          String value;
75          if(mapKey == null) {
76              value = (String)adaptor.getValue(targetObj);
77          } else {
78              value = (String)adaptor.getValueOfMap(mapKey, targetObj);
79          }
80          
81          value = Utils.trim(value, converterAnno);
82          value = Utils.getDefaultValueIfEmpty(value, converterAnno);
83          if(Utils.isNotEmpty(value)) {
84              cell.setCellValue(value);
85          } else {
86              cell.setCellType(Cell.CELL_TYPE_BLANK);
87          }
88          
89          return cell;
90      }
91      
92  }
