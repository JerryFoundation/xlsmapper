23  /**
24   * {@link Set}型を変換するためのConverter。
25   *
26   * @version 1.5
27   * @since 1.0
28   * @author T.TSUCHIE
29   *
30   */
31  @SuppressWarnings("rawtypes")
32  public class SetCellConverter extends AbstractCellConverter<Set> {
33      
34      @SuppressWarnings("unchecked")
35      @Override
36      public Set toObject(final Cell cell, final FieldAdaptor adaptor, final XlsMapperConfig config)
37              throws XlsMapperException {
38          
39          final ListCellConverter converter = new ListCellConverter();
40          final XlsArrayConverter anno = converter.getLoadingAnnotation(adaptor);
41          
42          Class<?> fieldClass = adaptor.getTargetClass();
43          
44          Class<?> itemClass = anno.itemClass();
45          if(itemClass == Object.class) {
46              itemClass = adaptor.getLoadingGenericClassType();
47          }
48          
49          final List<?> list = converter.toObject(cell, adaptor, config);
50          Set<?> set = (Set) Utils.convertListToCollection(list, (Class<Collection>)fieldClass, config.getBeanFactory());
51          return set;
52      }
53      
54      @Override
55      public Cell toCell(final FieldAdaptor adaptor, final Set targetValue, final Object targetBean,
56              final Sheet sheet, final int column, final int row,
57              final XlsMapperConfig config) throws XlsMapperException {
58          
59          final ListCellConverter converter = new ListCellConverter();
60          final XlsFormula formulaAnno = adaptor.getSavingAnnotation(XlsFormula.class);
61          final boolean primaryFormula = formulaAnno == null ? false : formulaAnno.primary();
62          
63          final XlsConverter converterAnno = adaptor.getSavingAnnotation(XlsConverter.class);
64          final XlsArrayConverter anno = converter.getSavingAnnotation(adaptor);
65          
66          Class<?> itemClass = anno.itemClass();
67          if(itemClass == Object.class) {
68              itemClass = adaptor.getSavingGenericClassType();
69          }
70          
71          final Cell cell = POIUtils.getCell(sheet, column, row);
72          
73          // セルの書式設定
74          if(converterAnno != null) {
75              POIUtils.wrapCellText(cell, converterAnno.wrapText());
76              POIUtils.shrinkToFit(cell, converterAnno.shrinkToFit());
77          }
78          
79          Set value = targetValue;
80          
81          // デフォルト値から値を設定する
82          if(Utils.isEmpty(value) && Utils.hasDefaultValue(converterAnno)) {
83              final List<?> list = converter.convertList(Utils.getDefaultValue(converterAnno), itemClass, converterAnno, anno, config);
84              value = new LinkedHashSet(list);
85          }
86          
87          if(Utils.isNotEmpty(value) && !primaryFormula) {
88              final boolean trim = (converterAnno == null ? false : converterAnno.trim()); 
89              final ItemConverter itemConverter = converter.getItemConverter(anno.itemConverterClass(), config);
90              final String cellValue = Utils.join(value, anno.separator(), anno.ignoreEmptyItem(), trim, itemConverter);
91              cell.setCellValue(cellValue);
92              
93          } else if(formulaAnno != null) {
94              Utils.setupCellFormula(adaptor, formulaAnno, config, cell, targetBean);
95              
96          } else {
97              cell.setCellType(Cell.CELL_TYPE_BLANK);
98          }
99          
100         return cell;
101     }
