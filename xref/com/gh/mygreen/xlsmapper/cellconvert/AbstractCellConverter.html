9   /**
10   * Converterの抽象クラス。

12   * 
13   * @version 0.5
14   * @author T.TSUCHIE

17  public abstract class AbstractCellConverter<T> implements CellConverter<T> {
18      
19      /**
20       * 型変換失敗したときの例外{@link TypeBindException}をスローします。
21       * @since 0.5
22       * @param error
23       * @param cell
24       * @param adaptor
25       * @param targetValue
26       * @return
27       */
28      public TypeBindException newTypeBindException(final Exception error, 
29              final Cell cell, final FieldAdaptor adaptor, final Object targetValue) {
30          
31          final String message = new StringBuilder()
32              .append(String.format("Fail conversion field value '%s' => type '%s'.",
33                      Utils.convertToString(targetValue), adaptor.getTargetClass()))
34              .append(String.format(" Cell '%s' map to '%s#%s'.", 
35                      Utils.formatCellAddress(cell), adaptor.getDeclaringClass().getName(), adaptor.getName()))
36              .toString();
37          
38          return new TypeBindException(error, message, adaptor.getTargetClass(), targetValue);
39          
40      }
41      
42      /**

49      public TypeBindException newTypeBindException(final Cell cell, final FieldAdaptor adaptor, final Object targetValue) {

55                      Utils.formatCellAddress(cell), adaptor.getDeclaringClass().getName(), adaptor.getName()))
